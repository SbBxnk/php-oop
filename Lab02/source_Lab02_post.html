        function calnum($a, $op, $b)
        {
            $c = 0;
            if ($op == "+")
                $c = $a + $b;
            elseif ($op == "-")
                $c = $a - $b;
            elseif ($op == "*")
                $c = $a * $b;
            elseif ($op == "/")
                $c = $a / $b;
            elseif ($op == "%")
                $c = $a % $b;
            elseif ($op == "^")
                $c = pow($a, $b);
            elseif ($op = "\\")
                $c = intdiv($a, $b);
            return $c;
        }
